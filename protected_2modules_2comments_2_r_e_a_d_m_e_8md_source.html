1 # yii-comments
2 
3 yii-comments is a module for Yii Framework which allows you to make any instance of CActiveRecord commentable.
4 This is a modification of the comments module available from <http://www.yiiframework.com/extension/comments-module/>
5 It is supposed that you have user module installed.
6 
7 # Changes
8 
9 Added Gravatar Support
10 Better Comment Management
11 Admin interface for editing comments
12 Admin interface for comments settings
13 Anchor for each comments (e.g. http://example.com/page/1#comment-12)
14 Better User Interaction
15 and others.

29 Extract the contents and move to `comments` folder inside modules.

31 Execute the following SQL queries to create table and insert default comment settings.
32 
33  CREATE TABLE IF NOT EXISTS `comment` (
34  `id` int(12) NOT NULL AUTO_INCREMENT,
35  `owner_id` int(11) NOT NULL,
36  `owner_name` varchar(50) NOT NULL,
37  `count` int(11) NOT NULL,
38  `parent_id` int(12) DEFAULT NULL,
39  `creator_id` int(12) DEFAULT NULL,
40  `user_name` varchar(128) DEFAULT NULL,
41  `user_email` varchar(128) DEFAULT NULL,
42  `comment_text` text,
43  `create_time` int(11) DEFAULT NULL,
44  `update_time` int(11) DEFAULT NULL,
45  `status` int(1) NOT NULL DEFAULT '0',
46  `link` text,
47  PRIMARY KEY (`id`)
48  );
49 
50  CREATE TABLE IF NOT EXISTS `comment_setting` (
51  `id` int(11) NOT NULL AUTO_INCREMENT,
52  `model` varchar(50) NOT NULL,
53  `registeredOnly` tinyint(1) NOT NULL DEFAULT '0',
54  `useCaptcha` tinyint(1) NOT NULL DEFAULT '0',
55  `allowSubcommenting` tinyint(1) NOT NULL DEFAULT '1',
56  `premoderate` tinyint(1) NOT NULL DEFAULT '0',
57  `isSuperuser` text,
58  `orderComments` enum('ASC','DESC') NOT NULL DEFAULT 'ASC',
59  `useGravatar` tinyint(1) NOT NULL DEFAULT '1',
60  PRIMARY KEY (`id`)
61  );
62 
63  INSERT INTO `comment_setting` (`id`, `model`, `registeredOnly`, `useCaptcha`, `allowSubcommenting`, `premoderate`, `isSuperuser`, `orderComments`, `useGravatar`) VALUES
64  (1, 'default', 0, 0, 1, 1, 'Yii::app()->getModule("user")->isAdmin()', 'ASC', 1);

87 # Usage:
88 
89 Add `$this->widget('Comments', array('model' => $model));` to the view of any model to make it commentable.
