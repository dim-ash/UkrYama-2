185  public static function run()
186  {
187  // check if the class was initialized
188  if (!self::$object)
189  throw new CHttpException('500', Yii::t('UserGroupsModule.cron','you didn\'t initialize the UGCron singleton'));
190  // array containing the loaded cronTables
191  $loaded_cron_tables = array();
192  foreach (self::$object->crons as $cron) {
193  // load the model cronTable model
194  if (!isset($loaded_cron_tables[$cron->cronTable])) {
195  $cTable = new $cron->cronTable;
196  $loaded_cron_tables[$cron->cronTable] = $cTable;
197  } else
198  $cTable = $loaded_cron_tables[$cron->cronTable];
199  // define the attributes for loading the cronjob parameters on the cronTable
200  $attributes = array('name' => $cron->name);
201 
202  // load the cron job from the cronTable
203  $cItem = $cTable->findByAttributes($attributes);
204  if ($cItem) {
205  // execute the cronjob if enough time has passed by
206  if ($cItem->last_occurrence <= date('Y-m-d', time() - (3600 * 24 * $cItem->lapse)).' 00:00:00') {
207  if ($cron->action === UGCronJob::DELETE) {
208 // Need to check if these users have holes added (checked in set criteria)
209 // this situation happens when hole is added with autoregistration funtion
210  $cron->_model->deleteAll($cron->criteria);
211  }
212  elseif ($cron->action === UGCronJob::UPDATE) {
213  $cron->_model->updateAll($cron->columns, $cron->criteria);
214  }
215  // update the cronTable
216  $cItem->last_occurrence = date('Y-m-d').' 00:00:00';
217  $cItem->save();
218  }
219  } else
220  self::$object->errors = array_merge(self::$object->errors, array($cron->name => Yii::t('UserGroupsModule.cron', 'Cronjob not installed')));
221  }
222  }

242  public final function checkIntegrity(UGCronJob $instance)
243  {
244  // check the integrity
245  if (!$instance->cronTable || !is_string($instance->cronTable))
246  throw new CHttpException('500', Yii::t('UserGroupsModule.cron', 'You didn\'t set the cron table name'));
247  $name = $instance->name;
248  if (!$name)
249  throw new CHttpException('500', Yii::t('UserGroupsModule.cron', 'You have to provide a name for the cronjob'));
250  $model = $instance->model;
251  if (!$model instanceof CActiveRecord)
252  return array($name => Yii::t('UserGroupsModule.cron', 'the model provided is not an instance of CActiveRecord'));
253  if (!$instance->criteria instanceof CDbCriteria)
254  return array($name => Yii::t('UserGroupsModule.cron', 'the provided criteria are not an instance of CDbCriteria'));
255  if ($instance->action !== self::DELETE && $instance->action !== self::UPDATE)
256  return array($name => Yii::t('UserGroupsModule.cron', 'you didn\'t provide a valid action'));
257  elseif ($instance->action === self::UPDATE && (!is_array($instance->columns) || !count($instance->columns)))
258  return array($name => Yii::t('UserGroupsModule.cron', 'you didn\'t set the columns to update'));
259 
260  // check if the cron is installed and install it if not
261  $cTable = new $instance->cronTable;
262  $attributes = array('name' => $instance->name);
263  // load the cron job from the cronTable
264  $cItem = $cTable->findByAttributes($attributes);
265  if ($cItem === NULL)
266  if (!$this->installCron($cTable, $name, $instance->lapse))
267  return array($name => Yii::t('UserGroupsModule.cron', 'could not install the module. Lapse must be an integer'));
268 
269 
270  return $model;
271  }
