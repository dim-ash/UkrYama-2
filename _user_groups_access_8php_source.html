172  private static function extractControllers($where, $module = false, $mode = 'dataProvider')
173  {
174  foreach (glob(Yii::getPathOfAlias($where) . "/*Controller.php") as $controller){
175  if ($module) {
176  if (DIRECTORY_SEPARATOR === '/') // fix for windows machines
177  self::$_moduleName = preg_replace('/^.*\/modules\/(.*)\/controllers.*$/', '$1', $controller);
178  else
179  self::$_moduleName = preg_replace('/^.*\\\modules\\\(.*)\\\controllers.*$/', '$1', $controller);
180  } else
181  self::$_moduleName = 'Basic';
182  $_controllerName = basename($controller, "Controller.php"); // TODO when stop supporting php 5.2 use lcfirst
183  $_controllerName{0} = strtolower($_controllerName{0});
184  self::$_controllerName = $_controllerName;
185 
186  $controller_class = ucfirst(self::$_controllerName.'Controller');
187 
188  // extract the value of permission controller inside the controller
189  if (!in_array($controller_class, self::$_alreadyIncluded)) {
190  // use reflectionClass if a controller with the same class name was not previously included
191  // add the controller class to the alreadyIncluded array
192  self::$_alreadyIncluded[] = $controller_class;
193  if (!in_array($controller_class, self::$_declaredClasses))
194  include($controller);
195 
196  $class = new ReflectionClass($controller_class);
197  if ($class->hasProperty('_permissionControl'))
198  $permissionControl = $class->getStaticPropertyValue('_permissionControl');
199  else
200  $permissionControl = NULL;
201  } else {
202  // parse the file if a controller with the same class name was previously included
203  // get the controller file content
204  $controller_file = file_get_contents($controller, false, NULL, 0);
205  // check if there is permissionControl inside it
206  if (strpos($controller_file, 'permissionControl') !== false) {
207  // get portion of the file containing permissionControl
208  $controller_file = substr($controller_file, strpos($controller_file, 'permissionControl'));
209  $controller_file = substr($controller_file, 0, strpos($controller_file, ';'));
210  $permissionControl = eval('return $'.$controller_file.';');
211  } else
212  $permissionControl = NULL;
213  }
214 
215  // check the value of permissionControl and skip this controller if necessary
216  if ($permissionControl === false || (count($permissionControl) === 1 && isset($permissionControl['label']) && $mode === 'dataProvider'))
217  continue;

219  if ($mode === 'dataProvider') {
220  self::$_rawData[] = array(
221  'id'=>NULL,
