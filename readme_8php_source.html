1 This module helps to add comments to any instance of CActiveRecord.
2 To add a comment to the model, you need to perform two steps.
3 First step - configure the module in app config:
4 
5 'modules'=>array(
6  ...
7  'comments'=>array(
8  //you may override default config for all connecting models
9  'defaultModelConfig' => array(
10  //only registered users can post comments
11  'registeredOnly' => false,
12  'useCaptcha' => false,
13  //allow comment tree
14  'allowSubcommenting' => true,
15  //display comments after moderation
16  'premoderate' => false,
17  //action for postig comment
18  'postCommentAction' => 'comments/comment/postComment',
19  //super user condition(display comment list in admin view and automoderate comments)
20  'isSuperuser'=>'false',
21  //order direction for comments
22  'orderComments'=>'DESC',
23  ),
24  //the models for commenting
25  'commentableModels'=>array(
26  //model with individual settings
27  'Citys'=>array(
28  'registeredOnly'=>true,
29  'useCaptcha'=>true,
30  'allowSubcommenting'=>false,
31  //config for create link to view model page(page with comments)
32  'pageUrl'=>array(
33  'route'=>'admin/citys/view',
34  'data'=>array('id'=>'city_id'),
35  ),
36  ),
37  //model with default settings
38  'ImpressionSet',
39  ),
40  //config for user models, which is used in application
41  'userConfig'=>array(
42  'class'=>'User',
43  'nameProperty'=>'username',
44  'emailProperty'=>'email',
45  ),
46  ),
47  ...
48 ),
49 
50 Second step - display ECommentListWidget in view for displaying commentable models
51 
52 $this->widget('comments.widgets.ECommentsListWidget', array(
53  'model' => $model,
54 ));
55
