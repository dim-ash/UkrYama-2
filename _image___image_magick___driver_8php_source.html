57  public function process($image, $actions, $dir, $file, $render = FALSE)
58  {
59  // We only need the filename
60  $image = $image['file'];
61 
62  // Unique temporary filename
63  $this->tmp_image = $dir.'k2img--'.sha1(time().$dir.$file).substr($file, strrpos($file, '.'));
64 
65  // Copy the image to the temporary file
66  copy($image, $this->tmp_image);
67 
68  // Quality change is done last
69  $quality = (int) CArray::remove('quality', $actions);
70 
71  // Use 95 for the default quality
72  empty($quality) and $quality = 95;
73 
74  // All calls to these will need to be escaped, so do it now
75  $this->cmd_image = escapeshellarg($this->tmp_image);
76  $this->new_image = ($render)? $this->cmd_image : escapeshellarg($dir.$file);
77 
78  if ($status = $this->execute($actions))
79  {
80  // Use convert to change the image into its final version. This is
81  // done to allow the file type to change correctly, and to handle
82  // the quality conversion in the most effective way possible.
83  if ($error = exec(escapeshellcmd($this->dir.'convert'.$this->ext).' -quality '.$quality.'% '.$this->cmd_image.' '.$this->new_image))
84  {
85  $this->errors[] = $error;
86  }
87  else
88  {
89  // Output the image directly to the browser
90  if ($render !== FALSE)
91  {
92  $contents = file_get_contents($this->tmp_image);
93  switch (substr($file, strrpos($file, '.') + 1))
94  {
95  case 'jpg':
96  case 'jpeg':
97  header('Content-Type: image/jpeg');
98  break;
99  case 'gif':
100  header('Content-Type: image/gif');
101  break;
102  case 'png':
103  header('Content-Type: image/png');
104  break;
105  }
106  echo $contents;
107  }
108  }
109  }
110 
111  // Remove the temporary image
112  unlink($this->tmp_image);
