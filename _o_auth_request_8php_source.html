30  public static function from_request($http_method=NULL, $http_url=NULL, $parameters=NULL) {
31  $scheme = (!isset($_SERVER['HTTPS']) || $_SERVER['HTTPS'] != "on")
32  ? 'http'

34  @$http_url or $http_url = $scheme .
35  '://' . $_SERVER['HTTP_HOST'] .

40 
41  // We weren't handed any parameters, so let's find the ones relevant to

44  // parsed parameter-list

47  $request_headers = OAuthUtil::get_headers();
48 
49  // Parse the query-string to find GET parameters
50  $parameters = OAuthUtil::parse_parameters($_SERVER['QUERY_STRING']);
51 
52  // It's a POST request of the proper content-type, so parse POST
53  // parameters and add those overriding any duplicates from GET
54  if ($http_method == "POST"

56  "application/x-www-form-urlencoded")

58  $post_data = OAuthUtil::parse_parameters(
59  file_get_contents(self::$POST_INPUT)
60  );

62  }
63

65  // and add those overriding any duplicates from GET or POST

67  $header_parameters = OAuthUtil::split_header(
68  $request_headers['Authorization']
69  );
70  $parameters = array_merge($parameters, $header_parameters);
