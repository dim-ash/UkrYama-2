278  $tree = array();
279  foreach ($data as $id => $node) {
280  $node->parent_id = $node->parent_id === null ? 0 : $node->parent_id;
281  if ($node->parent_id == $rootID) {
282  unset($data[$id]);
283  $node->childs = $this->buildTree($data, $node->id);
284  $tree[] = $node;
