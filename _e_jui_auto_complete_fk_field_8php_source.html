10 /*
11  * The EJuiAutoCompleteFkField extension renders a CJuiAutoComplete field plus supporting form fields for a FK field.
12  * Typically it is used for a model with a foreign key field to a parent table that has too many records
13  * for a drop-down list to be practical.
14  *
15  * For example it could be used in a Contact table, with a foreign key PostCodeId to a PostCode table
16  * with thousands of records, one for each city / postcode combination. The user would type the city name in the AutoCompleter,
17  * and the PostCodeId would be stored in the correct PostCodeId column; while the display attribute (e.g. City, Province) is shown
18  * in the form.
19  *
20  * The extension renders the following form objects:
21  * 1) the model field itself, which may optionally be hidden or visible
22  * 2) a hidden field that holds the description field of the FK record, for redisplay if the user
23  * fails to choose a value from the autoCompleter
24  * 3) the AutoComplete field itself, which also displays the existing value from the related record
25  * 4) a 'delete' icon to clear fields 1-3 above
26  * 5) javascript to tie everything together
27  *
28  * Typical usage:
29  * 1) unzip the extension into ../extensions/
30  *
31  * 2) make sure config/main.php has:
32  * <pre>
33  * ...
34  * import=>array(
35  * ...
36  * 'application.extensions.*',
37  * ...
38  * ),
39  * </pre>
40  *
41  * 3) ensure the relationship exists in the model:
42  * in Contacts.php (example):
43  * <pre>
44  * ...
45  * 'relations'=>array(
46  * 'Postcode'=>array('self::BELONGS_TO, 'PostCodes', 'PostCodeId'),
47  * ...
48  * </pre>
49  *
50  * 4) in the related table, optionally create a pseudo-attribute for display purposes
51  * in PostCodes.php (model) for example:
52  * <pre>
53  * ...
54  * public function getPostCodeAndProvince() {
55  * return $this->PostCodeId . ', ' . $this->Province;
56  * }
57  * </pre>
58  *
59  * 5) in the _form.php for the main record (e.g. Contacts)
60  * <pre>
61  * ...
62  * echo $form->labelEx($model, 'PostCodeId);
63  * $this->widget('EJuiAutoCompleteFkField', array(
64  * 'model'=>$model, // e.g. the Contact model (from CJuiInputWidget)
65  * 'attribute'=>'PostCodeId', // the FK field (from CJuiInputWidget)
66  * 'sourceUrl'=>'findPostCode', // name of the controller method to return the autoComplete data (see below) (from CJuiAutoComplete)
67  * 'showFKField'=>true, // defaults to false. set 'true' to display the FK value in the form with 'readonly' attribute.
68  * 'FKFieldSize=>15, // display size of the FK field. only matters if not hidden. defaults to 10
69  * 'relName'=>'Postcode', // the relation name defined above
70  * 'displayAttr'=>'PostCodeAndProvince', // attribute or pseudo-attribute to display
71  * 'autoCompleteLength'=>60, // length of the AutoComplete/display field, defaults to 50
72  * // any attributes of CJuiAutoComplete and jQuery JUI AutoComplete widget may also be defined. read the code and docs for all options
73  * 'options'=>array(
74  * 'minLength'=>3, // number of characters that must be typed before autoCompleter returns a value, defaults to 2
75  * ),
76  * ));
77  * echo $form->error($model, 'PostCodeId');
78  * ...
79  * </pre>
80  *
81  * 6) in the Controller for the model, create a method to return the autoComplete data.
82  * NOTE: make sure to give users the correct permission to execute this method, according to your security scheme
83  *
84  * in ContactsController.php (for example):
85  * </pre>
86  * // data provider for EJuiAutoCompleteFkField for PostCodeId field
87  * public function actionFindPostCode() {
88  * $q = $_GET['term'];
89  * if (isset($q)) {
90  * $criteria = new CDbCriteria;
91  * $criteria->condition = '...', //condition to find your data, using q1 as the parameter field
92  * $criteria->order = '...'; // correct order-by field
93  * $criteria->limit = ...; // probably a good idea to limit the results
94  * $criteria->params = array(':q' => trim($q) . '%'); // with trailing wildcard only; probably a good idea for large volumes of data
95  * $PostCodes = PostCodes::model()->findAll($criteria);
96  *
97  * if (!empty($PostCodes)) {
98  * $out = array();
99  * foreach ($PostCodes as $p) {
100  * $out[] = array(
101  * 'label' => $p->PostCodeAndProvince, // expression to give the string for the autoComplete drop-down
102  * 'value' => $p->PostCodeAndProvince, // probably the same expression as above
103  * 'id' => $p->PostCodeId, // return value from autocomplete
104  * );
105  * }
106  * echo CJSON::encode($out);
107  * Yii::app()->end();
108  * }
109  * }
110  * }
111  * </pre>
112  *
113  * 7) in the Controller loadModel() method, return the related record
114  * in ContactsController.php (for example)
115  * <pre>
116  * public function loadModel() {
117  * ...
118  * if (isset($_GET['id']))
119  * $this->_model=Contacts::model()->with('Postcode')->findbyPk($_GET['id']); // <==== NOTE 'with()'
120  * ...
121  * }
122  * </pre>
123  */
