152  {

178  $message->setBcc($bcc);

182  if ($sent == count($to) + count($cc) + count($bcc))
183  {
184  $evt->setResult(Swift_Events_SendEvent::RESULT_SUCCESS);
185  }
186  elseif ($sent > 0)
