10 /* The CAdvancedArBehavior extension adds up some functionality to the default
11  * possibilites of yii´s ActiveRecord implementation.
12  *
13  * To use this extension, just copy this file to your extensions/ directory,
14  * add 'import' => 'application.extensions.CAdvancedArBehavior', [...] to your
15  * config/main.php and add this behavior to each model you would like to
16  * inherit the new possibilities:
17  *
18  * public function behaviors(){
19  * return array( 'CAdvancedArBehavior' => array(
20  * 'class' => 'application.extensions.CAdvancedArBehavior'));
21  * }
22  *
23  *
24  * Better support of MANY_TO_MANY relations:
25  *
26  * When we have defined a MANY_MANY relation in our relations() function, we
27  * are now able to add up instances of the foreign Model on the fly while
28  * saving our Model to the Database. Let´s assume the following Relation:
29  *
30  * Post has:
31  * 'categories'=>array(self::MANY_MANY, 'Category',
32  * 'tbl_post_category(post_id, category_id)')
33  *
34  * Category has:
35  * 'posts'=>array(self::MANY_MANY, 'Post',
36  * 'tbl_post_category(category_id, post_id)')
37  *
38  * Now we can use the attribute 'categories' of our Post model to add up new
39  * rows to our MANY_MANY connection Table:
40  *
41  * $post = new Post();
42  * $post->categories = Category::model()->findAll();
43  * $post->save();
44  *
45  * This will save our new Post in the table Post, and in addition to this it
46  * updates our N:M-Table with every Category available in the Database.
47  *
48  * We can further limit the Objects given to the attribute, and can also go
49  * the other Way around:
50  *
51  * $category = new Category();
52  * $category->posts = array(5, 6, 7, 10);
53  * $caregory->save();
54  *
55  * We can pass Object instances like in the first example, or a list of
56  * integers that representates the Primary key of the Foreign Table, so that
57  * the Posts with the id 5, 6, 7 and 10 get´s added up to our new Category.
58  *
59  * 5 Queries will be performed here, one for the Category-Model and four for
60  * the N:M-Table tbl_post_category. Note that this behavior could be tuned
61  * further in the future, so only one query get´s executed for the MANY_MANY
62  * Table.
63  *
64  * We can also pass a _single_ object or an single integer:
65  *
66  * $category = new Category();
67  * $category->posts = Post::model()->findByPk(12);
68  * $category->posts = 12;
69  * $category->save();
70  */

88  public function writeManyManyTables()
89  {
90  Yii::trace('writing MANY_MANY data for '.get_class($this->owner),'system.db.ar.CActiveRecord');
91 
92  foreach($this->owner->relations() as $key => $relation)
93  {
94  if($relation['0'] == CActiveRecord::MANY_MANY) // ['0'] equals relationType
95  {
96  if(isset($this->owner->$key))
97  {
98  if(is_object($this->owner->$key) || is_numeric($this->owner->$key))
99  {
100  $this->executeManyManyEntry($this->makeManyManyDeleteCommand(
101  $relation[2],
102  $this->owner->{$this->owner->tableSchema->primaryKey}));
103  $this->executeManyManyEntry($this->owner->makeManyManyInsertCommand(
104  $relation[2],
105  (is_object($this->owner->$key))
106  ? $this->owner->$key->{$this->owner->$key->tableSchema->primaryKey}
107  : $this->owner->{$key}));
108  }
109  else //if (is_array($this->owner->$key) && $this->owner->$key != array())
110  {
111  $this->executeManyManyEntry($this->makeManyManyDeleteCommand(
112  $relation[2],
113  $this->owner->{$this->owner->tableSchema->primaryKey}));
114  foreach((array)$this->owner->$key as $foreignobject)
115  {
116  $this->executeManyManyEntry ($this->makeManyManyInsertCommand(
117  $relation[2],
118  (is_object($foreignobject))
119  ? $foreignobject->{$foreignobject->tableSchema->primaryKey}
120  : $foreignobject));
121  }
122  }
123  }
124  }
125  }

128  // We can't throw an Exception when this query fails, because it is possible
129  // that there is not row available in the MANY_MANY table, thus execute()
130  // returns 0 and the error gets thrown falsely.
131  public function executeManyManyEntry($query) {
132  Yii::app()->db->createCommand($query)->execute();
133  }
