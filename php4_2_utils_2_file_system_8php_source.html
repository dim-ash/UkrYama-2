39  {
40  if (is_null($path1)) {
41  $path1 = "";
42  }
43  if (is_null($path2)) {
44  $path2 = "";
45  }
46  if (!strlen($path2)) {
47  if (strlen($path1)) {
48  $_lastCharP1 = substr($path1, -1, 1);
49  if ($_lastCharP1 != "/" && $_lastCharP1 != "\\") {
50  $path1 .= DIRECTORY_SEPARATOR;

53  }
54  else {
55  $_firstCharP2 = substr($path2, 0, 1);

58  return $path2;
